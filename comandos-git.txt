git clone https://github.com/meind0201/umafactorial "clonar repositorio para uso local"
git config user.name nombre_usuario	"hacer login"
git config user.email email_usuario	"hacer login de tu email"
git add nombre_archivo			"añadir codificacion al area 'commit'"
git commit -m "mensaje"			"añadir modificaciones del area 'commit' al repositorio local poniendole una etiqueta"
git push origin 'rama'			"mandar los cambios del area local en esa rama al repositorio online"
git checkout (-b) 'rama'		"crear una rama con un nombre especifico y cambiar a esta.Si no ponemos -b solo cambiamos de rama"
git merge 'rama'			"mezcla rama actual con rama espeficicada"
git stash				"guarda contenido de rama en una 'pila' para guardarlo momentaneamente"
